CREATE OR REPLACE FUNCTION Date2EnrollYear(dDate IN DATE)
RETURN NUMBER
IS
	year NUMBER;
BEGIN
	SELECT TO_NUMBER(TO_CHAR(dDate, 'YYYY'))
	INTO year
	FROM dual;

	RETURN year;
END;
/

CREATE OR REPLACE FUNCTION Date2EnrollSemester(dDate IN DATE)
RETURN NUMBER
IS
	semester NUMBER;
BEGIN
	SELECT TO_NUMBER(TO_CHAR(dDate, 'MM'))
	INTO semester
	FROM dual;

	IF (semester >=11) THEN
		semester := 1; /* 다음 년도 */
	ELSIF (semester >= 5) THEN
		semester := 2;
	ELSE
		semester := 1;	
	END IF;

	RETURN semester;
END;
/
